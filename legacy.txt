  /*
    legacy.html
    - 古い iPad Safari でも読みやすいように、クライアント側は ES5 相当で記述。
    - let / const / => / template literal / spread / classList / fetch / Promise / CSS Grid / CSS変数 は不使用。
    - GAS側の getDashboardData() は既存のものをそのまま使う想定。

  (function () {
    var dashboardData = null;
    var lastLoadedAt = null;
    var reloadTimer = null;
    var nightState = 'auto';
    var displayMode = 'normal';
    var defaultSettings = {
      showSeconds: true,
      refreshMinutes: 5,
      maxTodayEvents: 5,
      maxTomorrowEvents: 3,
      maxTomorrowPeriods: 5,
      nightStart: '22:00',
      nightEnd: '06:00',
      showTermLabel: true,
      showInstallHint: true,
      burnInShiftMinutes: 5,
      burnInShiftAmount: 12,
      legacyTextSize: 'normal'
    };

    function el(id) {
      return document.getElementById(id);
    }

    function esc(value) {
      var text = String(value === null || value === undefined ? '' : value);
      return text
        .replace(/&/g, '&amp;')
        .replace(/</g, '&lt;')
        .replace(/>/g, '&gt;')
        .replace(/"/g, '&quot;')
        .replace(/'/g, '&#39;');
    }

    function pad2(value) {
      value = String(value);
      return value.length < 2 ? '0' + value : value;
    }

    function getSettings() {
      var result = {};
      var key;
      for (key in defaultSettings) {
        if (Object.prototype.hasOwnProperty.call(defaultSettings, key)) {
          result[key] = defaultSettings[key];
        }
      }
      if (dashboardData && dashboardData.settings) {
        for (key in dashboardData.settings) {
          if (Object.prototype.hasOwnProperty.call(dashboardData.settings, key)) {
            result[key] = dashboardData.settings[key];
          }
        }
      }
      return result;
    }

    function boolValue(value, defaultValue) {
      if (value === true || value === 'TRUE' || value === 'true' || value === '1' || value === 1) return true;
      if (value === false || value === 'FALSE' || value === 'false' || value === '0' || value === 0) return false;
      return defaultValue;
    }

    function formatClock(date) {
      return pad2(date.getHours()) + ':' + pad2(date.getMinutes());
    }

    function formatSeconds(date) {
      return pad2(date.getSeconds());
    }

    function formatFullTime(date) {
      return pad2(date.getHours()) + ':' + pad2(date.getMinutes()) + ':' + pad2(date.getSeconds());
    }

    function formatDate(date) {
      var weekdays = ['日', '月', '火', '水', '木', '金', '土'];
      return date.getFullYear() + '年' + (date.getMonth() + 1) + '月' + date.getDate() + '日 ' + weekdays[date.getDay()] + '曜日';
    }

    function formatDateTime(date) {
      return formatDate(date) + ' ' + formatFullTime(date);
    }

    function formatTimeFromMs(ms) {
      var date = new Date(Number(ms));
      if (!ms && ms !== 0) return '';
      return pad2(date.getHours()) + ':' + pad2(date.getMinutes());
    }

    function timeText(period) {
      if (!period) return '';
      return String(period.start || '') + '-' + String(period.end || '');
    }

    function toMinutes(hhmm) {
      var parts;
      if (!hhmm) return 0;
      parts = String(hhmm).split(':');
      if (parts.length < 2) return 0;
      return Number(parts[0]) * 60 + Number(parts[1]);
    }

    function isAutoNight(now, settings) {
      var current = now.getHours() * 60 + now.getMinutes();
      var start = toMinutes(settings.nightStart || '22:00');
      var end = toMinutes(settings.nightEnd || '06:00');
      if (start === end) return false;
      if (start < end) {
        return current >= start && current < end;
      }
      return current >= start || current < end;
    }

    function fontClass(settings) {
      var value = String(settings.legacyTextSize || settings.legacyFontSize || 'normal').toLowerCase();
      if (value === 'small' || value === '小') return 'font-small';
      if (value === 'large' || value === '大') return 'font-large';
      if (value === 'xlarge' || value === 'xl' || value === '特大') return 'font-xlarge';
      return '';
    }

    function modeLabel(mode) {
      if (mode === 'today') return '今日詳細';
      if (mode === 'tomorrow') return '明日詳細';
      return '通常';
    }

    function applyBodyClass(activeNight) {
      var settings = getSettings();
      var cls = 'mode-' + displayMode;
      var f = fontClass(settings);
      if (activeNight) cls = 'night ' + cls;
      if (f) cls += ' ' + f;
      document.body.className = cls;
      el('modeLine').innerHTML = '表示: ' + modeLabel(displayMode);
    }

    function updateNight() {
      var settings = getSettings();
      var active = false;
      if (nightState === 'on') active = true;
      if (nightState === 'off') active = false;
      if (nightState === 'auto') active = isAutoNight(new Date(), settings);
      applyBodyClass(active);
      el('nightButton').innerHTML = '夜間: ' + (nightState === 'auto' ? '自動' : (nightState === 'on' ? 'ON' : 'OFF'));
    }

    function cycleNight() {
      if (nightState === 'auto') {
        nightState = 'on';
      } else if (nightState === 'on') {
        nightState = 'off';
      } else {
        nightState = 'auto';
      }
      updateNight();
    }

    function cycleMode() {
      if (displayMode === 'normal') {
        displayMode = 'today';
      } else if (displayMode === 'today') {
        displayMode = 'tomorrow';
      } else {
        displayMode = 'normal';
      }
      updateNight();
      renderStatic();
    }

    function findCurrentAndNext(periods, nowMs) {
      var result = { current: null, next: null };
      var i;
      var p;
      if (!periods || !periods.length) return result;
      for (i = 0; i < periods.length; i++) {
        p = periods[i];
        if (p && nowMs >= Number(p.startMs) && nowMs < Number(p.endMs)) {
          result.current = p;
          break;
        }
      }
      for (i = 0; i < periods.length; i++) {
        p = periods[i];
        if (p && Number(p.startMs) > nowMs) {
          result.next = p;
          break;
        }
      }
      return result;
    }

    function renderMeta() {
      var today;
      var settings;
      var pieces = [];
      if (!dashboardData || !dashboardData.today) return;
      today = dashboardData.today;
      settings = getSettings();
      if (today.scheduleType) pieces.push(today.scheduleType);
      if (today.scheduleMemo) pieces.push(today.scheduleMemo);
      if (today.isVacation) pieces.push('休業');
      if (today.isSubstituteTimetableDay) pieces.push(today.timetableDayLabel + '曜授業');
      if (boolValue(settings.showTermLabel, true) && today.termLabel) pieces.push(today.termLabel);
      if (pieces.length) {
        el('metaLine').innerHTML = esc(pieces.join(' ｜ '));
      } else {
        el('metaLine').innerHTML = '&nbsp;';
      }
    }

    function renderMain() {
      var nowMs = (new Date()).getTime();
      var today;
      var found;
      var current;
      var next;
      var minutesLeft;
      var subject;
      var room;
      var status;
      if (!dashboardData || !dashboardData.today) return;
      today = dashboardData.today;
      found = findCurrentAndNext(today.periodsToday, nowMs);
      current = found.current;
      next = found.next;

      if (today.isVacation) {
        el('periodLine').innerHTML = '休業期間';
        el('subjectLine').innerHTML = esc(today.termMemo || today.scheduleMemo || today.termLabel || '授業はありません');
        el('remainingLine').innerHTML = '';
        el('nextBox').innerHTML = '<span class="muted">本日の時程はありません</span>';
        return;
      }

      if (current) {
        minutesLeft = Math.ceil((Number(current.endMs) - nowMs) / 60000);
        if (minutesLeft < 0) minutesLeft = 0;
        subject = current.subject || current.label || '';
        room = current.room ? ' <span class="sub">教室: ' + esc(current.room) + '</span>' : '';
        status = current.status ? ' <span class="status">' + esc(current.status) + '</span>' : '';
        el('periodLine').innerHTML = esc(current.label || '') + status;
        el('subjectLine').innerHTML = esc(subject) + room;
        el('remainingLine').innerHTML = '終了まで約' + minutesLeft + '分 ｜ ' + esc(timeText(current));
      } else if (next) {
        el('periodLine').innerHTML = '授業時間外';
        el('subjectLine').innerHTML = '次は ' + esc(next.label || '次の時限');
        el('remainingLine').innerHTML = esc(timeText(next));
      } else {
        el('periodLine').innerHTML = '本日の時程は終了';
        el('subjectLine').innerHTML = '';
        el('remainingLine').innerHTML = '';
      }

      if (next) {
        el('nextBox').innerHTML = makePeriodItem(next, false);
      } else {
        el('nextBox').innerHTML = '<span class="muted">次の時限はありません</span>';
      }
    }

    function makePeriodItem(period, active) {
      var html = '';
      var subject = period.subject || period.label || '';
      var label = period.label || '';
      var status = period.status ? ' <span class="status">' + esc(period.status) + '</span>' : '';
      var room = period.room ? ' / ' + period.room : '';
      var memo = period.memo ? '<span class="sub">' + esc(period.memo) + '</span>' : '';
      html += '<div class="item' + (active ? ' activeItem' : '') + '">';
      html += esc(period.start || '') + '-' + esc(period.end || '') + '　';
      html += esc(label) + status;
      html += '<span class="sub">' + esc(subject + room) + '</span>';
      html += memo;
      html += '</div>';
      return html;
    }

    function renderPeriods(boxId, periods, limit, nowMs) {
      var html = '';
      var i;
      var p;
      var active;
      var count = 0;
      if (!periods || !periods.length) {
        el(boxId).innerHTML = '<span class="muted">表示する時程はありません</span>';
        return;
      }
      for (i = 0; i < periods.length; i++) {
        if (limit && count >= limit) break;
        p = periods[i];
        if (!p) continue;
        active = nowMs && nowMs >= Number(p.startMs) && nowMs < Number(p.endMs);
        html += makePeriodItem(p, active);
        count++;
      }
      el(boxId).innerHTML = html || '<span class="muted">表示する時程はありません</span>';
    }

    function makeEventItem(ev) {
      var time = '';
      var title = '';
      var place = '';
      var memo = '';
      if (!ev) return '';
      if (ev.isAllDay) {
        time = '終日';
      } else if (ev.startMs) {
        time = formatTimeFromMs(ev.startMs);
        if (ev.endMs) time += '-' + formatTimeFromMs(ev.endMs);
      } else {
        time = '';
      }
      title = ev.title || ev.summary || '';
      place = ev.location ? '<span class="sub">場所: ' + esc(ev.location) + '</span>' : '';
      memo = ev.description ? '<span class="sub">' + esc(ev.description) + '</span>' : '';
      return '<div class="item">' + esc(time) + '　' + esc(title) + place + memo + '</div>';
    }

    function renderEventsToBox(boxId, events, limit, emptyText) {
      var html = '';
      var i;
      var max = Number(limit || 0);
      if (!events || !events.length || max <= 0) {
        el(boxId).innerHTML = '<span class="muted">' + esc(emptyText) + '</span>';
        return;
      }
      for (i = 0; i < events.length && i < max; i++) {
        html += makeEventItem(events[i]);
      }
      el(boxId).innerHTML = html || '<span class="muted">' + esc(emptyText) + '</span>';
    }

    function renderEvents() {
      var settings = getSettings();
      var events;
      var tomorrowEvents;
      var maxToday;
      var maxTomorrow;
      if (!dashboardData || !dashboardData.today) return;
      events = dashboardData.today.events || [];
      tomorrowEvents = dashboardData.tomorrow && dashboardData.tomorrow.events ? dashboardData.tomorrow.events : [];
      maxToday = Number(settings.maxTodayEvents || 5);
      maxTomorrow = Number(settings.maxTomorrowEvents || 3);
      renderEventsToBox('eventsBox', events, maxToday, '今日の予定はありません');
      renderEventsToBox('eventsDetailBox', events, events.length || maxToday, '今日の予定はありません');
      renderEventsToBox('tomorrowEventsBox', tomorrowEvents, maxTomorrow, '明日の予定はありません');
    }

    function renderInstallHint() {
      var settings = getSettings();
      var show = boolValue(settings.showInstallHint, true);
      var ua = window.navigator && window.navigator.userAgent ? window.navigator.userAgent : '';
      var isIOS = ua.indexOf('iPad') >= 0 || ua.indexOf('iPhone') >= 0 || ua.indexOf('iPod') >= 0;
      if (!show) {
        el('installHint').innerHTML = '';
        return;
      }
      if (isIOS) {
        el('installHint').innerHTML = 'ホーム画面に追加: Safariの共有ボタン → ホーム画面に追加';
      } else {
        el('installHint').innerHTML = 'ホーム画面に追加: ブラウザのメニューから追加';
      }
    }

    function updateBurnInShift() {
      var settings = getSettings();
      var minutes = Number(settings.burnInShiftMinutes || 0);
      var amount = Number(settings.burnInShiftAmount || 0);
      var bucket;
      var x = 0;
      var y = 0;
      if (!minutes || minutes < 1 || !amount || amount < 1) {
        el('wrap').style.left = '0px';
        el('wrap').style.top = '0px';
        return;
      }
      bucket = Math.floor((new Date()).getTime() / (minutes * 60000)) % 9;
      if (bucket === 1) { x = amount; y = 0; }
      if (bucket === 2) { x = amount; y = amount; }
      if (bucket === 3) { x = 0; y = amount; }
      if (bucket === 4) { x = -amount; y = amount; }
      if (bucket === 5) { x = -amount; y = 0; }
      if (bucket === 6) { x = -amount; y = -amount; }
      if (bucket === 7) { x = 0; y = -amount; }
      if (bucket === 8) { x = amount; y = -amount; }
      el('wrap').style.left = x + 'px';
      el('wrap').style.top = y + 'px';
    }

    function renderStatic() {
      var settings = getSettings();
      var nowMs = (new Date()).getTime();
      var tomorrowLimit;
      if (!dashboardData || !dashboardData.today) return;
      el('dateLine').innerHTML = esc(dashboardData.today.dateLabel || formatDate(new Date()));
      renderMeta();
      renderEvents();
      renderPeriods('todayPeriodsBox', dashboardData.today.periodsToday || [], 0, nowMs);
      if (dashboardData.tomorrow) {
        tomorrowLimit = displayMode === 'tomorrow' ? 0 : Number(settings.maxTomorrowPeriods || 5);
        renderPeriods('tomorrowBox', dashboardData.tomorrow.periodsToday || [], tomorrowLimit, 0);
      } else {
        el('tomorrowBox').innerHTML = '<span class="muted">明日のデータがありません</span>';
      }
      renderInstallHint();
      renderMain();
      updateNight();
      updateBurnInShift();
    }

    function updateClock() {
      var now = new Date();
      var settings = getSettings();
      el('clockMain').innerHTML = formatClock(now);
      if (boolValue(settings.showSeconds, true)) {
        el('clockSeconds').style.display = '';
        el('clockSeconds').innerHTML = formatSeconds(now);
      } else {
        el('clockSeconds').style.display = 'none';
      }
      if (!dashboardData || !dashboardData.today) {
        el('dateLine').innerHTML = formatDate(now);
      }
      renderMain();
      updateNight();
      updateBurnInShift();
    }

    function showError(message) {
      el('mainPanel').className = 'panel error';
      el('periodLine').innerHTML = '読み込み失敗';
      el('subjectLine').innerHTML = esc(message || 'データを取得できませんでした');
      el('remainingLine').innerHTML = '再読込を押してください';
    }

    function scheduleReload() {
      var settings = getSettings();
      var minutes = Number(settings.refreshMinutes || 5);
      if (!minutes || minutes < 1) minutes = 5;
      if (reloadTimer) window.clearTimeout(reloadTimer);
      reloadTimer = window.setTimeout(function () {
        refreshData(false);
      }, minutes * 60 * 1000);
    }

    function refreshData(manual) {
      el('mainPanel').className = 'panel';
      el('loadedLine').innerHTML = manual ? '再読込中...' : '更新中...';
      el('loadedDetailLine').innerHTML = '';

      if (typeof google === 'undefined' || !google.script || !google.script.run) {
        showError('google.script.run が使えません。Apps Script のWebアプリとして開いてください。');
        el('loadedLine').innerHTML = '更新失敗';
        return;
      }

      google.script.run
        .withSuccessHandler(function (data) {
          dashboardData = data || {};
          lastLoadedAt = new Date();
          renderStatic();
          el('loadedLine').innerHTML = '更新 ' + formatClock(lastLoadedAt);
          el('loadedDetailLine').innerHTML = '最終更新: ' + formatDateTime(lastLoadedAt);
          scheduleReload();
        })
        .withFailureHandler(function (error) {
          var msg = 'データを取得できませんでした';
          if (error && error.message) msg = error.message;
          showError(msg);
          el('loadedLine').innerHTML = '更新失敗 ' + formatClock(new Date());
          el('loadedDetailLine').innerHTML = '最終更新失敗: ' + formatDateTime(new Date());
          scheduleReload();
        })
        .getDashboardData();
    }

    function stopEvent(evt) {
      evt = evt || window.event;
      if (evt && evt.stopPropagation) evt.stopPropagation();
      if (evt) evt.cancelBubble = true;
    }

    function targetIsButton(evt) {
      var target = evt && (evt.target || evt.srcElement);
      var tag;
      if (!target) return false;
      tag = String(target.tagName || '').toUpperCase();
      return tag === 'BUTTON' || tag === 'A' || tag === 'INPUT' || tag === 'TEXTAREA' || tag === 'SELECT';
    }

    function init() {
      el('reloadButton').onclick = function (evt) { stopEvent(evt); refreshData(true); };
      el('nightButton').onclick = function (evt) { stopEvent(evt); cycleNight(); };
      el('modeButton').onclick = function (evt) { stopEvent(evt); cycleMode(); };
      el('wrap').onclick = function (evt) {
        if (targetIsButton(evt || window.event)) return;
        cycleMode();
      };
      updateNight();
      updateClock();
      window.setInterval(updateClock, 1000);
      refreshData(false);
    }

    if (window.attachEvent) {
      window.attachEvent('onload', init);
    } else {
      window.onload = init;
    }
  }());
  </script>
