    function loadData(options = {}) {
      const force = options.force === true;

      if (isLoading) return;
      setLoading(true);

      const handleSuccess = data => {
        // 新形式: { today, tomorrow, settings }。古い形式が返ってきても落ちないようにしておく。
        dashboard = data.today || data;
        tomorrowDashboard = data.tomorrow || null;
        dashboardSettings = normalizeClientSettings(data.settings || {});
        applySettings();
        applyDefaultModeFromSettings();
        lastUpdatedMs = Date.now();
        setLoading(false);
        render();
      };

      const handleFailure = error => {
        setLoading(false);
        document.getElementById('period').textContent = '取得エラー';
        document.getElementById('subject').textContent = error.message || String(error);
        updateFooterMeta();
      };

      if (force) {
        google.script.run
          .withSuccessHandler(() => {
            google.script.run
              .withSuccessHandler(handleSuccess)
              .withFailureHandler(handleFailure)
              .getDashboardData(DEVICE_ID);
          })
          .withFailureHandler(handleFailure)
          .clearDashboardCache();
        return;
      }

      google.script.run
        .withSuccessHandler(handleSuccess)
        .withFailureHandler(handleFailure)
        .getDashboardData(DEVICE_ID);
    }

    function setLoading(value) {
      isLoading = value;
      updateFooterMeta();
    }

    function render() {
      renderClock();
      updateNightMode();
      updateModeLabel();
      updateFooterMeta();

      if (!dashboard) return;

      const currentMode = getCurrentMode();
      const headerData = currentMode === 'tomorrow'
        ? tomorrowDashboard || dashboard
        : dashboard;

      document.getElementById('date').textContent =
        `${headerData.dateLabel}${makeScheduleText(headerData)}`;

      renderNormalView();
      renderDetailView();
    }

    function renderNormalView() {
      const now = Date.now();
      const periods = dashboard.periodsToday || [];
      const current = periods.find(p => now >= p.startMs && now < p.endMs);
      const next = periods.find(p => now < p.startMs);

      if (dashboard.isVacation) {
        document.getElementById('period').textContent = '休業期間';
        document.getElementById('subject').innerHTML = renderVacationInfo(dashboard);
        document.getElementById('remaining').textContent = '授業はありません';
      } else if (current) {
        document.getElementById('period').textContent = current.label;
        document.getElementById('subject').innerHTML = renderPeriodInfo(current);
        document.getElementById('remaining').textContent =
          `終了まで あと ${Math.ceil((current.endMs - now) / 60000)}分`;
      } else {
        document.getElementById('period').textContent = '授業時間外';
        document.getElementById('subject').textContent = '';
        document.getElementById('remaining').textContent = next
          ? `${formatTime(next.startMs)} から ${next.label}`
          : '今日の時程は終了';
      }

      if (dashboard.isVacation) {
        document.getElementById('next').innerHTML =
          `<div class="item muted">${renderVacationInfo(dashboard)}<br>次の時限はありません</div>`;
      } else if (next) {
        document.getElementById('next').innerHTML = `
          <div class="item">
            ${formatTime(next.startMs)}〜${formatTime(next.endMs)}<br>
            <strong>${escapeHtml(next.label)}</strong><br>
            <span class="muted">${renderPeriodInfo(next)}</span>
          </div>
        `;
      } else {
        document.getElementById('next').innerHTML =
          `<div class="item muted">次の時限はありません</div>`;
      }

      renderEvents();
      renderTomorrow();
    }

    function renderDetailView() {
      const currentMode = getCurrentMode();
      const detail = document.getElementById('detail');

      if (currentMode === 'normal') {
        detail.innerHTML = '';
        return;
      }

      const dayData = currentMode === 'tomorrow'
        ? tomorrowDashboard
        : dashboard;

      const title = currentMode === 'tomorrow'
        ? '明日詳細'
        : '今日詳細';

      detail.innerHTML = renderDayDetail(title, dayData, currentMode);
    }

    function renderDayDetail(title, dayData, targetMode) {
      if (!dayData) {
        return `<div class="detail-title">${escapeHtml(title)}</div><div class="item muted">情報がまだありません</div>`;
      }

      const now = Date.now();
      const periods = dayData.periodsToday || [];
      const events = dayData.events || [];

      const periodHtml = periods.length
        ? periods.map(p => {
            const isCurrent = targetMode === 'today' && now >= p.startMs && now < p.endMs;
            const isNext = targetMode === 'today' && !isCurrent && now < p.startMs && !periods.some(x => now < x.startMs && x.startMs < p.startMs);
            const marker = isCurrent
              ? '<span class="status">現在</span>'
              : isNext
                ? '<span class="status">次</span>'
                : '';

            return `
              <div class="detail-item ${isCurrent ? 'active' : ''}">
                <div class="detail-time">
                  ${escapeHtml(p.label)}<br>
                  ${formatTime(p.startMs)}〜${formatTime(p.endMs)}
                </div>
                <div class="detail-main">
                  ${marker}${renderPeriodInfo(p)}
                </div>
              </div>
            `;
          }).join('')
        : `<div class="item muted">${dayData.isVacation ? renderVacationInfo(dayData) : '授業はありません'}</div>`;

      const eventHtml = events.length
        ? events.map(e => {
            const time = e.isAllDay
              ? '終日'
              : `${formatTime(e.startMs)}〜${formatTime(e.endMs)}`;

            return `
              <div class="detail-item">
                <div class="detail-time">${time}</div>
                <div class="detail-main"><strong>${escapeHtml(e.title)}</strong></div>
              </div>
            `;
          }).join('')
        : `<div class="item muted">予定はありません</div>`;

      return `
        <div class="detail-title">${escapeHtml(title)}</div>
        <div class="detail-subtitle">${escapeHtml(dayData.dateLabel)}${escapeHtml(makeScheduleText(dayData))}</div>

        <div class="detail-section-title">時間割</div>
        ${periodHtml}

        <div class="detail-section-title">Google Calendar</div>
        ${eventHtml}
      `;
    }

    function renderClock() {
      const now = new Date();
      const hh = String(now.getHours()).padStart(2, '0');
      const mm = String(now.getMinutes()).padStart(2, '0');
      const ss = String(now.getSeconds()).padStart(2, '0');
      document.getElementById('clock-main').textContent = `${hh}:${mm}`;
      document.getElementById('clock-seconds').textContent = ss;
    }

    function renderEvents() {
      const now = Date.now();
      const events = (dashboard.events || [])
        .filter(e => e.endMs >= now || e.isAllDay)
        .slice(0, dashboardSettings.maxTodayEvents);

      const html = events.length
        ? events.map(e => {
            const time = e.isAllDay
              ? '終日'
              : `${formatTime(e.startMs)}〜${formatTime(e.endMs)}`;

            return `
              <div class="item">
                ${time}<br>
                <strong>${escapeHtml(e.title)}</strong>
              </div>
            `;
          }).join('')
        : `<div class="item muted">この後の予定はありません</div>`;

      document.getElementById('events').innerHTML = html;
    }

    function renderTomorrow() {
      const element = document.getElementById('tomorrow');
      if (!element) return;

      if (!tomorrowDashboard) {
        element.innerHTML = `<div class="item muted">明日の情報はまだありません</div>`;
        return;
      }

      const parts = [];
      parts.push(`
        <div class="item">
          <strong>${escapeHtml(tomorrowDashboard.dateLabel)}${escapeHtml(makeScheduleText(tomorrowDashboard))}</strong>
        </div>
      `);

      const periods = tomorrowDashboard.periodsToday || [];

      if (periods.length) {
        parts.push(periods.slice(0, dashboardSettings.maxTomorrowPeriods).map(p => `
          <div class="item">
            ${escapeHtml(p.label)}　${formatTime(p.startMs)}〜${formatTime(p.endMs)}<br>
            <span class="muted">${renderPeriodInfo(p)}</span>
          </div>
        `).join(''));
      } else {
        parts.push(`<div class="item muted">${tomorrowDashboard.isVacation ? renderVacationInfo(tomorrowDashboard) : '授業はありません'}</div>`);
      }

      const events = (tomorrowDashboard.events || []).slice(0, dashboardSettings.maxTomorrowEvents);

      if (events.length) {
        parts.push(events.map(e => {
          const time = e.isAllDay
            ? '終日'
            : `${formatTime(e.startMs)}〜${formatTime(e.endMs)}`;

          return `
            <div class="item">
              ${time}<br>
              <strong>${escapeHtml(e.title)}</strong>
            </div>
          `;
        }).join(''));
      }

      element.innerHTML = parts.join('');
    }

    function makeScheduleText(dayData) {
      if (!dayData) return '';

      const scheduleParts = [];

      if (
        dashboardSettings.showTermLabel &&
        dayData.termLabel &&
        dayData.termLabel !== '通年'
      ) {
        scheduleParts.push(dayData.termLabel);
      }

      if (dayData.scheduleType && dayData.scheduleType !== '通常') {
        scheduleParts.push(dayData.scheduleType);
      }

      if (dayData.isSubstituteTimetableDay) {
        scheduleParts.push(`${dayData.timetableDayLabel}曜授業`);
      }

      if (dayData.scheduleMemo) {
        scheduleParts.push(dayData.scheduleMemo);
      }

      return scheduleParts.length
        ? `｜${scheduleParts.join('・')}`
        : '';
    }

    function updateNightMode() {
      document.body.dataset.night = getEffectiveNightMode() ? 'true' : 'false';

    function isNightTime() {
      const now = new Date();
      const currentMinutes = now.getHours() * 60 + now.getMinutes();
      const startMinutes = parseClockMinutes(dashboardSettings.nightStart, 22 * 60);
      const endMinutes = parseClockMinutes(dashboardSettings.nightEnd, 6 * 60);

      if (startMinutes === endMinutes) return false;

      if (startMinutes < endMinutes) {
        return currentMinutes >= startMinutes && currentMinutes < endMinutes;
      }

      return currentMinutes >= startMinutes || currentMinutes < endMinutes;
    }

    function getEffectiveNightMode() {
      if (nightOverride === 'on') return true;
      if (nightOverride === 'off') return false;
      return isNightTime();
    }

    function loadNightOverride() {
      try {
        const value = localStorage.getItem(NIGHT_OVERRIDE_STORAGE_KEY);
        return ['auto', 'on', 'off'].includes(value) ? value : 'auto';
      } catch (error) {
        return 'auto';
      }
    }

    function saveNightOverride(value) {
      nightOverride = value;

      try {
        localStorage.setItem(NIGHT_OVERRIDE_STORAGE_KEY, value);
      } catch (error) {
        // 古い端末でlocalStorageが使えない場合は、その場限りの切替にする。
      }
    }

    function toggleNightOverride() {
      saveNightOverride(getEffectiveNightMode() ? 'off' : 'on');
      render();
    }

    function resetNightOverride() {
      saveNightOverride('auto');
      render();
    }

    function updateFooterMeta() {
      const updatedElement = document.getElementById('footer-updated');
      const refreshButton = document.getElementById('refresh-button');
      const deviceElement = document.getElementById('footer-device');
      const deviceSeparator = document.getElementById('device-separator');
      const nightToggle = document.getElementById('night-toggle');
      const nightReset = document.getElementById('night-reset');
      const wakeToggle = document.getElementById('wake-toggle');
      const wakeSeparator = document.getElementById('wake-separator');
      const fullscreenToggle = document.getElementById('fullscreen-toggle');
      const fullscreenSeparator = document.getElementById('fullscreen-separator');
      if (!updatedElement || !refreshButton || !nightToggle || !nightReset) return;

      updatedElement.textContent = lastUpdatedMs
        ? `更新 ${formatTime(lastUpdatedMs)}`
        : '更新 --:--';

      refreshButton.textContent = isLoading ? '読込中…' : '再読込';
      refreshButton.disabled = isLoading;

      if (deviceElement && deviceSeparator) {
        const deviceLabel = dashboardSettings.deviceLabel || dashboardSettings.device || '';
        deviceElement.textContent = deviceLabel ? `端末 ${deviceLabel}` : '';
        deviceElement.hidden = !deviceLabel;
        deviceSeparator.hidden = !deviceLabel;
      }

      const night = getEffectiveNightMode()
        ? '夜間ON'
        : '夜間OFF';

      nightToggle.textContent = nightOverride === 'auto'
        ? night
        : `${night}固定`;

      nightReset.hidden = nightOverride === 'auto';

      if (wakeToggle && wakeSeparator) {
        const enabled = Boolean(dashboardSettings.enableWakeLock);
        wakeToggle.hidden = !enabled;
        wakeSeparator.hidden = !enabled;

        if (enabled) {
          const supported = 'wakeLock' in navigator;
          wakeToggle.disabled = !supported;

          if (!supported) {
            wakeToggle.textContent = '画面ON不可';
          } else if (wakeLock) {
            wakeToggle.textContent = '画面ON中';
          } else if (wakeLockErrorMessage) {
            wakeToggle.textContent = '画面ON失敗';
          } else {
            wakeToggle.textContent = '画面ON';
          }
        }
      }

      if (fullscreenToggle && fullscreenSeparator) {
        const enabled = Boolean(dashboardSettings.enableFullscreen);
        fullscreenToggle.hidden = !enabled;
        fullscreenSeparator.hidden = !enabled;

        if (enabled) {
          const supported = isFullscreenSupported();
          const active = isFullscreenActive();
          fullscreenToggle.disabled = !supported;

          if (!supported) {
            fullscreenToggle.textContent = '全画面不可';
          } else if (active) {
            fullscreenToggle.textContent = '全画面解除';
          } else if (fullscreenErrorMessage) {
            fullscreenToggle.textContent = '全画面失敗';
          } else {
            fullscreenToggle.textContent = '全画面';
          }
        }
      }
    }

    function updateBurnInShift() {
      const minutes = Math.max(1, Number(dashboardSettings.burnInShiftMinutes) || DEFAULT_DASHBOARD_SETTINGS.burnInShiftMinutes);
      const amount = Math.max(0, Number(dashboardSettings.burnInShiftAmount) || DEFAULT_DASHBOARD_SETTINGS.burnInShiftAmount);
      const index = Math.floor(Date.now() / (minutes * 60 * 1000)) % BURN_IN_SHIFT_RATIOS.length;
      const [rx, ry] = BURN_IN_SHIFT_RATIOS[index];
      const x = Math.round(rx * amount);
      const y = Math.round(ry * amount);

      document.documentElement.style.setProperty('--burn-x', `${x}px`);
      document.documentElement.style.setProperty('--burn-y', `${y}px`);
    }

    async function toggleWakeLock() {
      wakeLockErrorMessage = '';

      if (!('wakeLock' in navigator)) {
        wakeLockRequested = false;
        wakeLockErrorMessage = '非対応';
        updateFooterMeta();
        return;
      }

      if (wakeLock) {
        wakeLockRequested = false;
        await releaseWakeLock();
        updateFooterMeta();
        return;
      }

      wakeLockRequested = true;
      await requestWakeLock();
      updateFooterMeta();
    }

    async function requestWakeLock() {
      if (!wakeLockRequested || !('wakeLock' in navigator) || document.visibilityState !== 'visible') return;

      try {
        wakeLock = await navigator.wakeLock.request('screen');
        wakeLockErrorMessage = '';
        wakeLock.addEventListener('release', () => {
          wakeLock = null;
          updateFooterMeta();
        });
      } catch (error) {
        wakeLock = null;
        wakeLockRequested = false;
        wakeLockErrorMessage = error?.message || '失敗';
      }
    }

    async function releaseWakeLock() {
      if (!wakeLock) return;

      const lock = wakeLock;
      wakeLock = null;

      try {
        await lock.release();
      } catch (error) {
        // すでに解放済みの場合は何もしない。
      }
    }


    function isFullscreenSupported() {
      const element = document.documentElement;
      return Boolean(
        element.requestFullscreen ||
        element.webkitRequestFullscreen ||
        element.msRequestFullscreen
      );
    }

    function isFullscreenActive() {
      return Boolean(
        document.fullscreenElement ||
        document.webkitFullscreenElement ||
        document.msFullscreenElement
      );
    }

    async function toggleFullscreen() {
      fullscreenErrorMessage = '';

      if (!isFullscreenSupported()) {
        fullscreenErrorMessage = '非対応';
        updateFooterMeta();
        return;
      }

      try {
        if (isFullscreenActive()) {
          await exitFullscreen();
        } else {
          await requestFullscreen();
        }
      } catch (error) {
        fullscreenErrorMessage = error?.message || '失敗';
      }

      updateFooterMeta();
    }

    async function requestFullscreen() {
      const element = document.documentElement;

      if (element.requestFullscreen) {
        await element.requestFullscreen();
        return;
      }

      if (element.webkitRequestFullscreen) {
        element.webkitRequestFullscreen();
        return;
      }

      if (element.msRequestFullscreen) {
        element.msRequestFullscreen();
      }
    }

    async function exitFullscreen() {
      if (document.exitFullscreen) {
        await document.exitFullscreen();
        return;
      }

      if (document.webkitExitFullscreen) {
        document.webkitExitFullscreen();
        return;
      }

      if (document.msExitFullscreen) {
        document.msExitFullscreen();
      }
    }

    function getInstallHintText() {

      const isIOS = /iPad|iPhone|iPod/.test(userAgent) || (navigator.platform === 'MacIntel' && navigator.maxTouchPoints > 1);

      if (isIOS) {
        return '共有ボタン → 「ホーム画面に追加」で、SafariのUIを減らして開けます。';

      if (isAndroid) {
        return 'ブラウザのメニュー → 「ホーム画面に追加」または「アプリをインストール」で開けます。';

      return 'ブラウザのメニューから「ホーム画面に追加」または「アプリをインストール」を選べます。';
    }

    function isStandaloneDisplay() {
      return window.matchMedia('(display-mode: standalone)').matches || window.navigator.standalone === true;

      if (autoRefreshTimerId) {
        clearInterval(autoRefreshTimerId);

    function normalizeBoolean(value, fallback) {
      if (value === true || value === false) return value;

      const text = String(value || '').trim().toLowerCase();
      if (['true', 'yes', 'y', 'on', '1'].includes(text)) return true;
      if (['false', 'no', 'n', 'off', '0'].includes(text)) return false;

      return fallback;
    }

    function clampNumber(value, min, max, fallback) {
      const number = Number(value);
      if (!Number.isFinite(number)) return fallback;
      return Math.min(max, Math.max(min, number));
    }

    function normalizeClockText(value, fallback) {
      const text = String(value || '').trim();
      return /^\d{1,2}:\d{2}$/.test(text) ? text.padStart(5, '0') : fallback;
    }

    function normalizeTextSetting(value, fallback, maxLength) {
      const text = String(value || '').trim();
      return text ? text.slice(0, maxLength) : fallback;
    }

    function normalizeColorText(value, fallback) {
      const text = String(value || '').trim();
      return /^#[0-9a-fA-F]{6}$/.test(text) ? text : fallback;
    }

    function normalizeUrlText(value, fallback) {
      const text = String(value || '').trim();
      if (!text) return fallback || '';
      if (text.length > 4096) return fallback || '';

      if (/^https:\/\/.+\.(png|ico|jpg|jpeg|svg|webp)(\?.*)?$/i.test(text)) {
        return text;
      }

      if (/^data:image\/(png|jpeg|webp|svg\+xml);/i.test(text)) {
        return text;
      }

      return fallback || '';
    }

    function parseClockMinutes(value, fallback) {
      const text = normalizeClockText(value, '');
      if (!text) return fallback;

      const [hour, minute] = text.split(':').map(Number);
      if (!Number.isFinite(hour) || !Number.isFinite(minute)) return fallback;
      if (hour < 0 || hour > 23 || minute < 0 || minute > 59) return fallback;

      return hour * 60 + minute;
    }

    function formatTime(ms) {
      const d = new Date(ms);
      return `${String(d.getHours()).padStart(2, '0')}:${String(d.getMinutes()).padStart(2, '0')}`;
    }

    function escapeHtml(str) {
      return String(str)
        .replaceAll('&', '&amp;')
        .replaceAll('<', '&lt;')
        .replaceAll('>', '&gt;')
        .replaceAll('"', '&quot;')
        .replaceAll("'", '&#039;');
    }

    function renderVacationInfo(dayData) {
      const label = dayData?.termLabel || '休業期間';
      const memo = dayData?.termMemo || dayData?.scheduleMemo || '';
      return memo
        ? `${escapeHtml(label)}<div class="memo">${escapeHtml(memo)}</div>`
        : escapeHtml(label);
    }

    function renderPeriodInfo(period) {
      const subject = period.subject || '予定なし';

      const status = period.status && period.status !== '変更'
        ? `<span class="status">${escapeHtml(period.status)}</span>`
        : '';

      const room = period.room
        ? `<span class="room">@${escapeHtml(period.room)}</span>`
        : '';

      const memo = period.memo
        ? `<div class="memo">${escapeHtml(period.memo)}</div>`
        : '';

      return `${escapeHtml(subject)}${status}${room}${memo}`;
    }
